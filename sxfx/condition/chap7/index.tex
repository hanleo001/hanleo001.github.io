\subsubsection{变限积分}\label{ux53d8ux9650ux79efux5206}}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  若\(f\in R[a,b]\)，则\(\int_a^xf(t)dt \in C[a,b]\)
\item
  若\(f\)在\(x_0\)处连续，则\(\int_a^xf(t)dt\)在\(x_0\)初可导
\end{enumerate}
